## 📈 数据监控面板

### 每日检查（建议）

```bash
# 1. 今日使用情况
sqlite3 test_logs.db "
SELECT 
  DATE(timestamp) as 日期，
  COUNT(DISTINCT session_id) as 用户数，
  COUNT(*) as 使用次数，
  AVG(response_time_ms)/1000 as 平均响应 (秒)
FROM test_logs 
WHERE DATE(timestamp)=DATE('now')
GROUP BY DATE(timestamp);
"

# 2. 反馈统计
sqlite3 test_logs.db "
SELECT 
  COUNT(*) as 反馈数，
  AVG(rating) as 平均分，
  SUM(CASE WHEN accuracy='accurate' THEN 1 ELSE 0 END) as 准确数
FROM user_feedback 
WHERE DATE(timestamp)=DATE('now');
"

# 3. 最新反馈
sqlite3 test_logs.db "
SELECT rating, feedback_text, timestamp 
FROM user_feedback 
ORDER BY timestamp DESC 
LIMIT 5;
"
```

### 周报生成

```bash
# 本周汇总
sqlite3 test_logs.db "
SELECT 
  COUNT(DISTINCT session_id) as 总用户数，
  COUNT(*) as 总使用次数，
  AVG(response_time_ms)/1000 as 平均响应 (秒)，
  COUNT(DISTINCT DATE(timestamp)) as 活跃天数
FROM test_logs 
WHERE DATE(timestamp) >= DATE('now', '-7 days');
"

# 本周反馈
sqlite3 test_logs.db "
SELECT 
  COUNT(*) as 反馈数，
  AVG(rating) as 平均分
FROM user_feedback 
WHERE DATE(timestamp) >= DATE('now', '-7 days');
"
```
