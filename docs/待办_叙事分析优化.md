### 修复方案 B：前端自动分段

**文件：** `frontend/App.vue`

**修改 `formatNarrative()` 方法：**

**修改前：**
```javascript
formatNarrative(text) {
  if (!text) return ''
  // ...类型检查...
  return text
    .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
    .replace(/\*(.+?)\*/g, '<em>$1</em>')
    .replace(/ /g, '<br>')
}
```

**修改后：**
```javascript
formatNarrative(text) {
  if (!text) return ''
  // ...类型检查...
  
  let html = text
    // 粗体
    .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
    // 斜体
    .replace(/\*(.+?)\*/g, '<em>$1</em>')
  
  // 智能分段：按句号 + 空格/换行分割，每 2-3 句为一段
  const sentences = html.split(/([。！？.!?]\s*)/)
  const paragraphs = []
  for (let i = 0; i < sentences.length; i += 4) { // 每 2 句为一段（句号 + 内容算 2 个元素）
    const paragraph = sentences.slice(i, i + 4).join('').trim()
    if (paragraph) {
      paragraphs.push(`<p>${paragraph}</p>`)
    }
  }
  
  return paragraphs.join('\n')
}
```
